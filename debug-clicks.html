6. Finde Elemente die √ºber den Cards liegen:

// Finde das erste weather-card Element
const card = document.querySelector('.weather-card');
if (card) {
  const rect = card.getBoundingClientRect();
  const centerX = rect.left + rect.width / 2;
  const centerY = rect.top + rect.height / 2;

  const elementAtPoint = document.elementFromPoint(centerX, centerY);
  console.log('Card position:', rect);
  console.log('Element at card center:', elementAtPoint);
  console.log('Is it the card itself?', elementAtPoint === card);

  // Liste alle Elemente an diesem Punkt
  const elements = [];
  let el = elementAtPoint;
  while (el) {
    elements.push({
      tag: el.tagName,
      id: el.id,
      classes: el.className,
      zIndex: getComputedStyle(el).zIndex,
      pointerEvents: getComputedStyle(el).pointerEvents
    });
    el = el.parentElement;
  }
  console.table(elements);
}
